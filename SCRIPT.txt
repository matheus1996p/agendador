CREATE TABLE CONF_AGENDAMENTO (
    ID           INTEGER NOT NULL,
    BLOQUEADOS   VARCHAR(200),
    HORAINICIAL  VARCHAR(80),
    HORAFINAL    VARCHAR(80),
    INTERVALO    INTEGER,
    DOMINGO      INTEGER,
    SEGUNDA      INTEGER,
    TERCA        INTEGER,
    QUARTA       INTEGER,
    QUINTA       INTEGER,
    SEXTA        INTEGER,
    SABADO       INTEGER
);

CREATE TABLE REGISTRAR_HORARIO (
    ID          INTEGER,
    CODITEM      INTEGER,
    CPF         VARCHAR(200),
    PLACA       VARCHAR(100),
    HORARIO     VARCHAR(80),
    STATUS      INTEGER,
    DATA        DATE,
    QUANTIDADE  INTEGER,
    DESCRICAO   VARCHAR(200),
    DOCUMENTO   VARCHAR(100)
);

CREATE GENERATOR GEN_REGISTRAR_HORARIO

CREATE GENERATOR GEN_REGISTRAR_AGENDAMENTO

CREATE TRIGGER TRIGGER_REGISTRAR_HORARIO for REGISTRAR_HORARIO
BEFORE INSERT position 0
AS
BEGIN
new.id = gen_id("GEN_REGISTRAR_HORARIO",1);
END;

CREATE TRIGGER TRIGGER_REGISTRAR_AGENDAMENTO for CONF_AGENDAMENTO
BEFORE INSERT position 0
AS
BEGIN
new.id = gen_id("GEN_REGISTRAR_AGENDAMENTO",1);
END;
